```typescript
/**
 * proxy-helpers.ts
 *
 * All proxy protocol functions for the GSD decision proxy agent.
 * Implements: discuss_choice reception, discuss_answer sending,
 * DECISIONS.md logging, orchestrator-side choice sending + timeout/fallback.
 */

import type {
  PeerId,
  Message,
  DiscussChoicePayload,
  DiscussAnswerPayload,
  PollMessagesResponse,
} from "../../shared/types.ts";

// --- Configuration ---

const BROKER_PORT = process.env.CLAUDE_PEERS_PORT ?? "7899";
const BROKER_URL = `http://127.0.0.1:${BROKER_PORT}`;

// --- Internal broker communication ---

async function brokerFetch<T>(path: string, body: unknown): Promise<T> {
  const res = await fetch(`${BROKER_URL}${path}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  if (!res.ok) {
    const err = await res.text();
    throw new Error(`Broker error (${path}): ${res.status} ${err}`);
  }
  return res.json() as Promise<T>;
}
```

**Exported functions (7 total):**

1. `pollForChoices(myId: PeerId)` — PRXY-01
   - Calls `brokerFetch<PollMessagesResponse>("/poll-messages", { id: myId })`
   - Filters for `msg_type === "discuss_choice"`
   - Returns `Array<{ id: number; from_id: PeerId; choicePayload: DiscussChoicePayload }>` by JSON.parsing each message's `payload` field
   - Does NOT ACK messages (caller decides when to ACK after processing)

2. `parseChoicePayload(raw: string): DiscussChoicePayload` — PRXY-01 helper
   - `JSON.parse(raw) as DiscussChoicePayload`
   - Validates required fields exist: `phase_number`, `question`, `options`, `recommended`
   - Throws Error with message `"Invalid DiscussChoicePayload: missing required field '{fieldName}'"` if any required field is undefined

3. `buildAnswerPayload(phaseNumber: number, chosen: string, reasoning: string): DiscussAnswerPayload` — PRXY-02 helper
   - Returns `{ phase_number: phaseNumber, chosen, reasoning }`

4. `sendAnswer(myId: PeerId, orchestratorId: PeerId, payload: DiscussAnswerPayload): Promise<void>` — PRXY-02
   - Calls `brokerFetch("/send-message", { from_id: myId, to_id: orchestratorId, text: "Phase " + payload.phase_number + " decision: " + payload.chosen, msg_type: "discuss_answer", payload })`

5. `appendDecision(decisionsPath: string, phaseNumber: number, question: string, chosen: string, reasoning: string): Promise<void>` — PRXY-04
   - Uses `Bun.file(decisionsPath).exists()` to check if file exists
   - If file does not exist: writes `"# Autonomous Run Decisions\n\n"` + entry
   - If file exists: reads existing content with `await Bun.file(decisionsPath).text()`, appends entry
   - Entry format (exactly this):
     ```
     ## Phase {phaseNumber}
     **Question:** {question}
     **Chosen:** {chosen}
     **Reasoning:** {reasoning}
     **Timestamp:** {new Date().toISOString()}

     ```
   - Uses `await Bun.write(decisionsPath, existing + entry)` (append-only, entries accumulate at bottom)

6. `sendDiscussChoice(myId: PeerId, proxyId: PeerId, choice: DiscussChoicePayload): Promise<void>` — PRXY-01 (orchestrator side)
   - Calls `brokerFetch("/send-message", { from_id: myId, to_id: proxyId, text: "Phase " + choice.phase_number + " choice: " + choice.question, msg_type: "discuss_choice", payload: choice })`

7. `waitForAnswer(myId: PeerId, phaseNumber: number, timeoutMs: number = 60_000): Promise<DiscussAnswerPayload | null>` — PRXY-05
   - Polls `brokerFetch<PollMessagesResponse>("/poll-messages", { id: myId })` every 2 seconds
   - Looks for message where `msg_type === "discuss_answer"` AND parsed payload `phase_number === phaseNumber`
   - When found: ACKs the message via `brokerFetch("/ack-message", { message_ids: [answer.id] })`, returns parsed `DiscussAnswerPayload`
   - Also ACKs and discards any `discuss_answer` messages where `phase_number !== phaseNumber` (stale answer cleanup per pitfall 1 in RESEARCH.md)
   - If `Date.now() >= deadline` (where `deadline = Date.now() + timeoutMs` computed at start): returns `null`
   - Uses `await new Promise(r => setTimeout(r, 2_000))` between polls

**ACK helper (also exported for proxy agent use):**

8. `ackMessages(messageIds: number[]): Promise<void>`
   - Calls `brokerFetch("/ack-message", { message_ids: messageIds })`
   - Exported so the proxy agent can ACK discuss_choice messages after processing
