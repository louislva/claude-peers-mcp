```typescript
const BROKER_PORT = parseInt(process.env.CLAUDE_PEERS_PORT ?? "7899", 10);
const BROKER_URL = `http://127.0.0.1:${BROKER_PORT}`;

async function brokerFetch<T>(path: string, body?: unknown): Promise<T> {
  const opts: RequestInit = body
    ? {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body),
      }
    : {};
  const res = await fetch(`${BROKER_URL}${path}`, {
    ...opts,
    signal: AbortSignal.timeout(3000),
  });
  if (!res.ok) {
    throw new Error(`${res.status}: ${await res.text()}`);
  }
  return res.json() as Promise<T>;
}
```

Create `tui/render.ts` (~150 LOC) with raw ANSI escape code primitives. Zero external dependencies.

**Screen buffer management:**
- `enterAltScreen()`: writes `\x1b[?1049h` to enter alternate screen buffer
- `exitAltScreen()`: writes `\x1b[?1049l` to restore original screen
- `clearScreen()`: writes `\x1b[2J\x1b[H` to clear and home cursor
- `hideCursor()`: writes `\x1b[?25l`
- `showCursor()`: writes `\x1b[?25h`

**Cursor control:**
- `moveTo(row: number, col: number)`: writes `\x1b[{row};{col}H` (1-based)
- `write(text: string)`: `process.stdout.write(text)`

**Color system (ANSI 256-color only, not true color):**
- `fg(color: number): string` — returns `\x1b[38;5;{color}m`
- `bg(color: number): string` — returns `\x1b[48;5;{color}m`
- `resetStyle(): string` — returns `\x1b[0m`
- `bold(): string` — returns `\x1b[1m`
- `dim(): string` — returns `\x1b[2m`
- Named color constants object `C` with ANSI 256 values:
  ```
  C.bg = 233        // dark background
  C.bgLight = 234   // slightly lighter bg
  C.purple = 99     // accent
  C.green = 34      // success
  C.yellow = 214    // warning
  C.red = 203       // error
  C.blue = 75       // peer IDs
  C.dimGray = 238   // timestamps
  C.gray = 240      // secondary
  C.bright = 254    // headers
  C.text = 250      // normal text
  ```

**Box-drawing and layout:**
- `drawBox(row: number, col: number, width: number, height: number, title?: string)`: draws a box using Unicode box-drawing characters (single-line: `\u250c \u2510 \u2514 \u2518 \u2500 \u2502`). Title rendered in bright text centered in top border.
- `drawHLine(row: number, col: number, width: number)`: draws horizontal line using `\u2500`
- `badge(text: string, colorCode: number): string` — returns `[{text}]` with fg color set and reset. Example: badge("DONE", C.green) returns `\x1b[38;5;34m[DONE]\x1b[0m`
- `truncate(text: string, maxLen: number): string` — truncates with `...` if text exceeds maxLen
- `padRight(text: string, width: number): string` — pads with spaces to width

**Terminal size:**
- `getTermSize(): { rows: number; cols: number }` — reads `process.stdout.rows` and `process.stdout.columns`

All functions that produce output write directly via `process.stdout.write()`. String-returning functions (fg, bg, badge, truncate, etc.) return escape code strings for composition.

Create `tui/input.ts` (~80 LOC) with raw stdin keypress parsing. Zero external dependencies.

**Types:**
```typescript
export interface KeyEvent {
  name: string;       // "q", "1", "2", ..., "6", "tab", "shift-tab", "up", "down",
                      // "j", "k", "e", "w", "enter", "?", "escape", "unknown"
  raw: Buffer;        // original bytes
  ctrl: boolean;      // Ctrl modifier detected
  shift: boolean;     // Shift modifier detected (for shift-tab)
}

export type KeyHandler = (key: KeyEvent) => void;
```

**Functions:**
- `startInput(handler: KeyHandler): void`
  - Sets `process.stdin.setRawMode(true)` and `process.stdin.resume()`
  - Listens on `process.stdin` `'data'` event
  - Parses raw byte sequences into KeyEvent objects:
    - Single printable chars: name = the char itself (e.g., "q", "1", "e", "w", "j", "k", "?")
    - `\x1b[A` = "up", `\x1b[B` = "down", `\x1b[C` = "right", `\x1b[D` = "left"
    - `\x09` = "tab"
    - `\x1b[Z` = "shift-tab"
    - `\x0d` = "enter"
    - `\x1b` alone (after 50ms timeout to distinguish from escape sequences) = "escape"
    - `\x03` = Ctrl+C: name = "c", ctrl = true
    - Anything else: name = "unknown"
  - Calls handler(keyEvent) for each parsed key
- `stopInput(): void`
  - Sets `process.stdin.setRawMode(false)` and `process.stdin.pause()`
  - Removes the data listener

The escape sequence timeout (50ms) distinguishes bare Escape from multi-byte escape sequences like arrow keys. If `\x1b` arrives and no more bytes follow within 50ms, emit "escape". If more bytes arrive, parse as a sequence.

Export KeyEvent, KeyHandler, startInput, stopInput as named exports.

```typescript
const BROKER_PORT = parseInt(process.env.CLAUDE_PEERS_PORT ?? "7899", 10);
export const BROKER_URL = `http://127.0.0.1:${BROKER_PORT}`;

export async function brokerFetch<T>(path: string, body?: unknown): Promise<T> {
  const opts: RequestInit = body
    ? {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body),
      }
    : {};
  const res = await fetch(`${BROKER_URL}${path}`, {
    ...opts,
    signal: AbortSignal.timeout(3000),

    throw new Error(`${res.status}: ${await res.text()}`);
  }
  return res.json() as Promise<T>;
}
```

Also export a convenience wrapper for safe polling (used by tabs that poll on intervals):

```typescript
export async function safeFetch<T>(path: string, body?: unknown): Promise<T | null> {
  try {
    return await brokerFetch<T>(path, body);
  } catch {
    return null;
  }
}
```

And export a broker health check:

```typescript
export async function isBrokerUp(): Promise<boolean> {
  try {
    await brokerFetch<{ status: string }>("/health");
    return true;
  } catch {
    return false;
  }
}
```
